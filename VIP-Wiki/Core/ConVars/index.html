  • Переменные (ConVar`s)

Консольные переменные

cfg/vip/VIP_Core.cfg 
// Добавить пункт "Управление VIP" в админ-меню.
// -
// Default: "1"
sm_vip_add_item_to_admin_menu "1"

// Флаг админа, необходимый чтобы иметь доступ к управлению VIP-игроками.
// -
// Default: "z"
sm_vip_admin_flag "z"

// Автоматически открывать VIP-меню при входе (0 - Выключено, 1 - Включено)
// -
// Default: "0"
// Minimum: "0.000000"
// Maximum: "1.000000"
sm_vip_auto_open_menu "0"

// Удалять VIP-игроков у которых истек срок:
//      -1 - Не удалять
//      0 - Удалять сразу
//      Больше 0 - Количество дней, по истечению которых удалять
// -
// Default: "1"
// Minimum: "-1.000000"
sm_vip_delete_expired "1"

// Режим отображения недоступных функций в вип меню (0 - Сделать пункты неактивными, 1 - Скрывать пункты)
// -
// Default: "0"
// Minimum: "0.000000"
// Maximum: "1.000000"
sm_vip_hide_no_access_items "0"

// Вести ли лог logs/VIP_Logs.log (0 - Выключено, 1 - Включено)
// -
// Default: "1"
// Minimum: "0.000000"
// Maximum: "1.000000"
sm_vip_logs_enable "1"

// Команды для вызова VIP-меню (разделять ;)
// -
// Default: "vip;sm_vip;sm_vipmenu"
sm_vip_menu_commands "vip;sm_vip;sm_vipmenu"

// ID сервера при приспользовании MySQL базы данных
// -
// Default: "0"
// Minimum: "0.000000"
sm_vip_server_id "0"

// Задержка перед установкой привилегий при возрождении игрока
// -
// Default: "1.0"
// Minimum: "0.100000"
sm_vip_spawn_delay "1.0"

// Формат времени
//      0 - Секунды
//      1 - Минуты
//      2 - Часы
//      3 - Дни
// -
// Default: "0"
// Minimum: "0.000000"
// Maximum: "3.000000"
sm_vip_time_mode "0"

// Обновлять имена VIP-игроков в базе данных при входе(0 - Выключено, 1 - Включено)
// -
// Default: "1"
// Minimum: "0.000000"
// Maximum: "1.000000"
sm_vip_update_name "1"

// Статус функций по-умолчанию (0 - Выключены, 1 - Включены)
// -
// Default: "1"
// Minimum: "0.000000"
// Maximum: "1.000000"
sm_vip_features_default_status "1"
